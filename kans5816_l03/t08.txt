SELECT m.memberSurname, m.memberForename,
        (SELECT COUNT(*)
         FROM memberBroad mb
         WHERE mb.memberBroadBroadId = m.memberId) AS broadCount
FROM member m
WHERE (SELECT COUNT(*)
       FROM memberBroad mb
       WHERE mb.memberBroadBroadId = m.memberId) >= 8
ORDER BY m.memberSurname, m.memberForename
