SELECT m.memberSurname, m.memberForename,
    
    (SELECT COUNT(*)
     FROM pub p
     WHERE p.pubMemberId = m.memberId 
       AND p.pubPubType = 'b') AS books,
     
    (SELECT COUNT(*)
     FROM pub p
     WHERE p.pubMemberId = m.memberId 
       AND p.pubPubType = 'a') AS articles, 

    (SELECT COUNT(*)
     FROM pub p
     WHERE p.pubMemberId = m.memberId 
       AND p.pubPubType = 'c') AS papers

FROM member m
ORDER BY m.memberSurname, m.memberForename
